-- –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è —Ä–∞–∑–º–µ—Ä–∞ –º–æ–¥–µ–ª–∏
local function getModelSize(model)
    local parts = model:GetDescendants()
    local min, max = Vector3.new(math.huge, math.huge, math.huge), Vector3.new(-math.huge, -math.huge, -math.huge)
    
    for _, part in pairs(parts) do
        if part:IsA("BasePart") then
            local size = part.Size
            local cframe = part.CFrame
            local corners = {
                cframe * Vector3.new(-size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, -size.Y/2, size.Z/2),
                cframe * Vector3.new(-size.X/2, size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, size.Y/2, size.Z/2),
                cframe * Vector3.new(size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(size.X/2, -size.Y/2, size.Z/2),
                cframe * Vector3.new(size.X/2, size.Y/2, -size.Z/2),
                cframe * Vector3.new(size.X/2, size.Y/2, size.Z/2)
            }
            for _, corner in pairs(corners) do
                min = Vector3.new(math.min(min.X, corner.X), math.min(min.Y, corner.Y), math.min(min.Z, corner.Z))
                max = Vector3.new(math.max(max.X, corner.X), math.max(max.Y, corner.Y), math.max(max.Z, corner.Z))
            end
        end
    end
    
    if min == Vector3.new(math.huge, math.huge, math.huge) then
        return Vector3.new(2, 2, 2) -- –§–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Ä–∞–∑–º–µ—Ä, –µ—Å–ª–∏ –Ω–µ—Ç —á–∞—Å—Ç–µ–π
    end
    return max - min + Vector3.new(0.2, 0.2, 0.2) -- –î–æ–±–∞–≤–ª—è–µ–º –Ω–µ–±–æ–ª—å—à–æ–π –æ—Ç—Å—Ç—É–ø
end

-- –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –ø–æ–∑–∏—Ü–∏–∏ –æ–±—ä–µ–∫—Ç–∞ (–¥–ª—è –º–æ–¥–µ–ª–µ–π ‚Äî —Ü–µ–Ω—Ç—Ä –≥–∞–±–∞—Ä–∏—Ç–æ–≤)
local function getObjectPosition(obj)
    if obj:IsA("BasePart") then
        return obj.Position
    elseif obj:IsA("Model") then
        local cframe = obj:GetBoundingBox()
        return cframe.Position
    end
    return Vector3.new(0, 0, 0)
end

-- –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è ESP –∏ –Ω–∞–¥–ø–∏—Å–∏
local function createESP(obj)
    if not obj or espElements[obj] then return end -- –ü—Ä–æ–ø—É—Å–∫–∞–µ–º, –µ—Å–ª–∏ —É–∂–µ –µ—Å—Ç—å ESP –∏–ª–∏ –æ–±—ä–µ–∫—Ç –Ω–µ–¥–µ–π—Å—Ç–≤–∏—Ç–µ–ª–µ–Ω

    -- –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ä–∞–∑–º–µ—Ä –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç —Ç–∏–ø–∞ –æ–±—ä–µ–∫—Ç–∞
    local espSize
    if obj:IsA("BasePart") then
        espSize = obj.Size + Vector3.new(0.2, 0.2, 0.2)
    elseif obj:IsA("Model") then
        espSize = getModelSize(obj)
    else
        return -- –ü—Ä–æ–ø—É—Å–∫–∞–µ–º –Ω–µ–ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞–µ–º—ã–µ —Ç–∏–ø—ã
    end

    -- –û–ø—Ä–µ–¥–µ–ª—è–µ–º –ø–∞—Ä–∞–º–µ—Ç—Ä—ã –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç —Ç–∏–ø–∞ –æ–±—ä–µ–∫—Ç–∞ (Gold –∏–ª–∏ Shark)
    local isGold = obj.Name == "Gold"
    local isShark = obj.Name == "Shark"

    -- –µ—Å–ª–∏ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∞—è —Ñ—É–Ω–∫—Ü–∏—è –≤—ã–∫–ª—é—á–µ–Ω–∞ ‚Äî –Ω–µ —Å–æ–∑–¥–∞—ë–º ESP
    if (isGold and not showGold) or (isShark and not showShark) then
        return
    end

    local defaultEspColor = isGold and Color3.new(1, 1, 0) or Color3.new(0, 0.5, 1) -- –ñ–µ–ª—Ç—ã–π –¥–ª—è Gold, –≥–æ–ª—É–±–æ–π –¥–ª—è Shark
    local defaultLabelColor = defaultEspColor -- –¶–≤–µ—Ç –Ω–∞–¥–ø–∏—Å–∏ –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é
    local defaultLabelText = isGold and "Gold" or "ü¶à" -- –ù–∞–¥–ø–∏—Å—å –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é
    local defaultLabelSize = isGold and UDim2.new(0, 15, 0, 15) or UDim2.new(0, 17.5, 0, 17.5) -- –†–∞–∑–º–µ—Ä: –º–µ–Ω—å—à–µ –¥–ª—è Gold –∏ Shark

    -- –°–æ–∑–¥–∞–µ–º BoxHandleAdornment (ESP)
    local esp = Instance.new("BoxHandleAdornment")
    esp.Adornee = obj
    esp.AlwaysOnTop = true
    esp.ZIndex = 0
    esp.Size = espSize
    esp.Transparency = 0.5
    esp.Color3 = defaultEspColor
    esp.Parent = coreGui

    -- –°–æ–∑–¥–∞–µ–º BillboardGui –¥–ª—è –Ω–∞–¥–ø–∏—Å–∏
    local billboard = Instance.new("BillboardGui")
    billboard.Adornee = obj
    billboard.Size = isGold and UDim2.new(0, 50, 0, 40) or defaultLabelSize -- –î–ª—è Gold –±–æ–ª—å—à–µ, —á—Ç–æ–±—ã –≤–º–µ—Å—Ç–∏—Ç—å –¥–∏—Å—Ç–∞–Ω—Ü–∏—é
    billboard.StudsOffset = Vector3.new(0, espSize.Y + 1, 0) -- –ù–∞–¥ –æ–±—ä–µ–∫—Ç–æ–º
    billboard.AlwaysOnTop = true

    -- –ù–∞–¥–ø–∏—Å—å —Å –∏–º–µ–Ω–µ–º (Gold –∏–ª–∏ ü¶à)
    local textLabel = Instance.new("TextLabel")
    textLabel.Size = isGold and UDim2.new(1, 0, 0.5, 0) or UDim2.new(1, 0, 1, 0) -- –î–ª—è Gold –ø–æ–ª–æ–≤–∏–Ω–∞ –≤—ã—Å–æ—Ç—ã
    textLabel.Position = isGold and UDim2.new(0, 0, 0, 0) or UDim2.new(0, 0, 0, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.Text = defaultLabelText
    textLabel.TextColor3 = defaultLabelColor
    textLabel.TextScaled = true
    textLabel.Font = Enum.Font.SourceSansBold
    textLabel.Parent = billboard

    -- –ù–∞–¥–ø–∏—Å—å —Å –¥–∏—Å—Ç–∞–Ω—Ü–∏–µ–π (—Ç–æ–ª—å–∫–æ –¥–ª—è Gold)
    local distanceLabel
    if isGold then
        distanceLabel = Instance.new("TextLabel")
        distanceLabel.Size = UDim2.new(1, 0, 0.5, 0)
        distanceLabel.Position = UDim2.new(0, 0, 0.5, 0) -- –ü—Ä—è–º–æ –ø–æ–¥ –æ—Å–Ω–æ–≤–Ω–æ–π –Ω–∞–¥–ø–∏—Å—å—é

    billboard.Parent = coreGui

    -- –°–æ—Ö—Ä–∞–Ω—è–µ–º ESP –∏ BillboardGui
    espElements[obj] = {esp = esp, billboard = billboard, distanceLabel = distanceLabel, defaultSize = espSize, defaultLabelSize = defaultLabelSize, defaultLabelText = defaultLabelText, isGold = isGold, defaultEspColor = defaultEspColor, defaultLabelColor = defaultLabelColor}

    -- –ü—Ä–æ–≤–µ—Ä—è–µ–º, –µ—Å–ª–∏ –æ–±—ä–µ–∫—Ç –∏—Å—á–µ–∑–∞–µ—Ç

        if not obj:IsDescendantOf(workspace) then
            if espElements[obj] then
                if espElements[obj].esp then pcall(function() espElements[obj].esp:Destroy() end) end
                if espElements[obj].billboard then pcall(function() espElements[obj].billboard:Destroy() end) end
                if espElements[obj].distanceLabel then pcall(function() elements.distanceLabel:Destroy() end) end
                espElements[obj] = nil
            end
        end
    end)
end

-- –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–∏—Å–∫–∞ –≤—Å–µ—Ö –æ–±—ä–µ–∫—Ç–æ–≤ "Gold" –∏ "Shark"
local function findAndApplyESP()
    for _, obj in pairs(workspace:GetDescendants()) do
        if (obj.Name == "Gold" or obj.Name == "Shark") and (obj:IsA("BasePart") or obj:IsA("Model")) then
            createESP(obj)
        end
    end
end

getgenv()._LC_findAndApplyESP = findAndApplyESP

-- –ü–µ—Ä–≤–æ–Ω–∞—á–∞–ª—å–Ω—ã–π –ø–æ–∏—Å–∫ (—É—á–∏—Ç—ã–≤–∞–µ—Ç showGold/showShark)
findAndApplyESP()

-- –û—Ç—Å–ª–µ–∂–∏–≤–∞–Ω–∏–µ –Ω–æ–≤—ã—Ö –æ–±—ä–µ–∫—Ç–æ–≤ (–µ—Å–ª–∏ Gold –∏–ª–∏ Shark —Å–ø–∞–≤–Ω—è—Ç—Å—è –≤ –∏–≥—Ä–µ)
workspace.DescendantAdded:Connect(function(obj)
    if (obj.Name == "Gold" or obj.Name == "Shark") and (obj:IsA("BasePart") or obj:IsA("Model")) then
        createESP(obj)
    end
end)

-- –ì–ª–æ–±–∞–ª—å–Ω—ã–π Heartbeat –¥–ª—è –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è –≤—Å–µ—Ö ESP –∏ –æ—á–∏—Å—Ç–∫–∏
runService.Heartbeat:Connect(function()
    if not localPlayer.Character or not localPlayer.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    local playerPos = localPlayer.Character.HumanoidRootPart.Position
    
    for obj, elements in pairs(espElements) do
        if not obj:IsDescendantOf(workspace) then
            if elements.esp then pcall(function() elements.esp:Destroy() end) end
            if elements.billboard then pcall(function() elements.billboard:Destroy() end) end
            if elements.distanceLabel then pcall(function() elements.distanceLabel:Destroy() end) end
            espElements[obj] = nil
            continue
        end
        
        local objPos = getObjectPosition(obj)
        local distance = (playerPos - objPos).Magnitude
        
        if elements.isGold then
            -- –ï—Å–ª–∏ Gold —Å–µ–π—á–∞—Å –≤—ã–∫–ª—é—á–µ–Ω ‚Äî —É–¥–∞–ª—è–µ–º –∏ –ø—Ä–æ–ø—É—Å–∫–∞–µ–º
            if not showGold then
                if elements.esp then pcall(function() elements.esp:Destroy() end) end
                if elements.billboard then pcall(function() elements.billboard:Destroy() end) end
                if elements.distanceLabel then pcall(function() elements.distanceLabel:Destroy() end) end
                espElements[obj] = nil
                continue
            end

            -- –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –¥–ª—è Gold
            if elements.distanceLabel then
                elements.distanceLabel.Text = string.format("%.1f studs", distance)
                if distance <= 150 then
                    elements.esp.Color3 = Color3.new(1, 0, 0) -- –ö—Ä–∞—Å–Ω—ã–π
                    elements.esp.Size = elements.defaultSize * 1.1 -- –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º –Ω–∞ 10%
                    elements.distanceLabel.TextColor3 = Color3.new(1, 0, 0)
                    local tl = elements.billboard:FindFirstChild("TextLabel")
                    if tl then tl.TextColor3 = Color3.new(1, 0, 0) end
                else
                    elements.esp.Color3 = elements.defaultEspColor -- –ñ–µ–ª—Ç—ã–π
                    elements.esp.Size = elements.defaultSize
                    elements.distanceLabel.TextColor3 = elements.defaultLabelColor
                    local tl = elements.billboard:FindFirstChild("TextLabel")
                    if tl then tl.TextColor3 = elements.defaultLabelColor end
                end
            end
        else
            -- Shark
            if not showShark then
                if elements.esp then pcall(function() elements.esp:Destroy() end) end
                if elements.billboard then pcall(function() elements.billboard:Destroy() end) end
                if elements.distanceLabel then pcall(function() elements.distanceLabel:Destroy() end) end
                espElements[obj] = nil
                continue
            end

            local textLabel = elements.billboard:FindFirstChild("TextLabel")
            if distance <= 100 then
                if textLabel then
                    textLabel.Text = "‚ô¶Ô∏è" -- –ú–µ–Ω—è–µ–º –Ω–∞ —Ä–æ–º–±
                    textLabel.TextColor3 = Color3.new(1, 0, 0) -- –ö—Ä–∞—Å–Ω—ã–π
                end
                elements.billboard.Size = UDim2.new(0, 41.67, 0, 41.67) -- –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º –≤ 1.5 —Ä–∞–∑–∞ –º–µ–Ω—å—à–µ –æ—Ç 62.5
                elements.esp.Size = elements.defaultSize * 1.1 -- –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º ESP –Ω–∞ 10%
            else
                if textLabel then
                    textLabel.Text = elements.defaultLabelText -- –í–æ–∑–≤—Ä–∞—â–∞–µ–º ü¶à
                    textLabel.TextColor3 = elements.defaultLabelColor -- –ì–æ–ª—É–±–æ–π
                end
                elements.billboard.Size = elements.defaultLabelSize
                elements.esp.Size = elements.defaultSize
            end
            elements.esp.Color3 = elements.defaultEspColor -- ESP –≤—Å–µ–≥–¥–∞ –≥–æ–ª—É–±–æ–π
        end
    end
end)
